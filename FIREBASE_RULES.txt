rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Shared company income data - all approved users can read and write
    match /company/{document=**} {
      allow read: if isApproved();
      allow write: if isApproved();
    }
    
    // User approval whitelist - users can only read their own approval status
    match /approvedUsers/{uid} {
      allow read: if request.auth.uid == uid;
      allow create: if false;
      allow update: if false;
      allow delete: if false;
    }
    
    // Helper function to check if user is approved
    function isApproved() {
      return request.auth != null && 
             exists(/databases/$(database)/documents/approvedUsers/$(request.auth.uid)) &&
             get(/databases/$(database)/documents/approvedUsers/$(request.auth.uid)).data.approved == true;
    }
  }
}
